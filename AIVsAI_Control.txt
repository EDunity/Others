                        ListGameObject_CubeD[i - 30 - b].transform.GetComponent<Rigidbody>().AddExplosionForce(Random.Range(0, 500), Mark_Cube.transform.position + new Vector3(5.5f, 15.5f, 0) + new Vector3(0, 5, 5), 100, 0);
                        ListGameObject_CubeD[i - 30 - b].transform.GetComponent<Rigidbody>().AddExplosionForce(Random.Range(0, 500), Mark_Cube.transform.position + new Vector3(5.5f, 15.5f, 0) + new Vector3(0, 0, 5), 100, 0);
                        ListGameObject_CubeD[i - 30 - b].transform.GetComponent<Rigidbody>().AddExplosionForce(Random.Range(0, 500), Mark_Cube.transform.position + new Vector3(5.5f, 15.5f, 0) + new Vector3(0, -5, 5), 100, 0);
                        ListGameObject_CubeD[i - 30 - b].transform.GetComponent<Rigidbody>().AddExplosionForce(Random.Range(0, 500), Mark_Cube.transform.position + new Vector3(5.5f, 15.5f, 0) + new Vector3(0, -10, 5), 100, 0);
                        break;
                }

                a += 1;

                if (a == 10)
                {
                    a = 0;
                    b += 4;
                }
            }
        }
    }


    //フェードイン
    void FadingIn_VsAI()
    {
        Image_Image1.color = new Color(Image_Image1.color.r, Image_Image1.color.g, Image_Image1.color.b, Image_Image1.color.a + 0.01f);

        Invoke("FadingIn_VsAI", 0.01f);

        if (Image_Image1.color.a >= 0.99f)
        {
            Image_Image1.color = new Color(Image_Image1.color.r, Image_Image1.color.g, Image_Image1.color.b, 1);

            CancelInvoke("FadingIn_VsAI");
        }
    }
    //フェードアウト
    void FadingOut_VsAI()
    {
        Image_Image1.color = new Color(Image_Image1.color.r, Image_Image1.color.g, Image_Image1.color.b, Image_Image1.color.a - 0.01f);

        Invoke("FadingOut_VsAI", 0.01f);

        if (Image_Image1.color.a <= 0.01f)
        {
            Image_Image1.color = new Color(Image_Image1.color.r, Image_Image1.color.g, Image_Image1.color.b, 0);

            CancelInvoke("FadingOut_VsAI");
        }
    }
